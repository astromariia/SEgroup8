yolo_local_eval.py

This module provides a local (non-Colab) workflow for:

  1. (Optionally) training a YOLOv8 model using the Ultralytics API.
  2. Evaluating image-level class accuracy on a validation set, where an image is considered correct if all ground-truth classes appear in the model's predictions (bounding box locations are ignored).
  3. Reporting per-class accuracy statistics.

It is written to be compatible with pydoc, so all executable logic is contained in functions and guarded under the usual if __name__ == "__main__": block.

 93def read_gt_classes(label_path: Path) -> Set[int]:
 94    """
 95    Read ground-truth classes from a YOLO-format label file.
 96
 97    The expected format for each line in the label file is:
 98        class_id x_center y_center width height
 99
100    Only the `class_id` (the first value on each line) is used here.
101
102    Parameters
103    ----------
104    label_path : Path
105        Path to the YOLO label file (.txt).
106
107    Returns
108    -------
109    Set[int]
110        A set of integer class IDs present in this annotation file.
111        Returns an empty set if the file does not exist or contains no labels.
112    """
113    if not label_path.exists():
114        return set()
115
116    classes: Set[int] = set()
117    with open(label_path, "r", encoding="utf-8") as f:
118        for line in f:
119            parts = line.strip().split()
120            if not parts:
121                continue
122            # The first value is the class ID
123            cls_id = int(float(parts[0]))
124            classes.add(cls_id)
125
126    return classes

129def evaluate_image_level_accuracy(
130    model: YOLO,
131    images_dir: Path,
132    labels_dir: Path,
133    class_names: list,
134    conf_threshold: float = 0.25,
135) -> Tuple[float, Dict[int, float]]:
136    """
137    Evaluate image-level class accuracy over a directory of images.
138
139    For each image, this function:
140      1. Loads the ground-truth classes from the YOLO label file.
141      2. Uses the model to predict bounding boxes for the image.
142      3. Extracts the unique set of predicted class IDs.
143      4. Counts the image as correct if *all* ground-truth classes are present
144         in the predicted set (bounding box locations are ignored).
145
146    Additionally, it tracks per-class accuracy: for each class,
147    "correct" means the class was present in the ground truth for an image
148    and also appeared at least once in the model's predictions.
149
150    Parameters
151    ----------
152    model : YOLO
153        A loaded Ultralytics YOLO model ready to perform `.predict()`.
154    images_dir : Path
155        Directory containing the validation images (jpg/png/jpeg).
156    labels_dir : Path
157        Directory containing YOLO-format annotation files matching
158        the images in `images_dir`.
159    class_names : list
160        List of class names, indexed by numeric class ID.
161    conf_threshold : float, optional
162        Confidence threshold for predictions, by default 0.25.
163
164    Returns
165    -------
166    Tuple[float, Dict[int, float]]
167        A tuple `(overall_accuracy, per_class_accuracy)` where:
168
169        - `overall_accuracy` is the fraction of images for which
170          all ground-truth classes appeared in the predictions.
171        - `per_class_accuracy` is a dictionary mapping class ID to
172          the probability that the class is predicted at least once,
173          given that it appears in the ground truth for that image.
174    """
175    # Collect all image paths in the directory
176    image_paths = sorted(
177        list(images_dir.glob("*.jpg"))
178        + list(images_dir.glob("*.png"))
179        + list(images_dir.glob("*.jpeg"))
180    )
181
182    total_images = 0
183    correct_images = 0
184
185    # Per-class statistics
186    class_totals: Dict[int, int] = defaultdict(int)
187    class_correct: Dict[int, int] = defaultdict(int)
188
189    for img_path in image_paths:
190        label_path = labels_dir / (img_path.stem + ".txt")
191        gt_classes = read_gt_classes(label_path)
192
193        # Skip images that have no ground-truth labels
194        if len(gt_classes) == 0:
195            continue
196
197        # Run the model on this single image
198        results = model.predict(
199            source=str(img_path),
200            conf=conf_threshold,
201            verbose=False,
202        )
203
204        r = results[0]
205
206        # Extract predicted class IDs (if any)
207        if r.boxes is not None and len(r.boxes) > 0:
208            pred_classes = set(r.boxes.cls.cpu().numpy().astype(int).tolist())
209        else:
210            pred_classes = set()
211
212        # Overall image-level accuracy
213        total_images += 1
214        if gt_classes.issubset(pred_classes):
215            correct_images += 1
216
217        # Per-class statistics: if class appears in GT, did we predict it?
218        for cls_id in gt_classes:
219            class_totals[cls_id] += 1
220            if cls_id in pred_classes:
221                class_correct[cls_id] += 1
222
223    # Compute final overall accuracy
224    overall_accuracy = (
225        correct_images / total_images if total_images > 0 else 0.0
226    )
227
228    # Compute per-class accuracy
229    per_class_accuracy: Dict[int, float] = {}
230    for cls_id, total in class_totals.items():
231        if total > 0:
232            per_class_accuracy[cls_id] = class_correct[cls_id] / total
233        else:
234            per_class_accuracy[cls_id] = 0.0
235
236    # Print a summary to the console
237    print(f"Images evaluated: {total_images}")
238    print(f"Images with all GT classes predicted: {correct_images}")
239    print(f"Image-level class accuracy: {overall_accuracy * 100:.2f}%")
240
241    print("\nPer-class accuracy (GT present  predicted at least once):\n")
242    for cls_id in sorted(class_totals.keys()):
243        total = class_totals[cls_id]
244        correct = class_correct[cls_id]
245        acc = per_class_accuracy[cls_id]
246        name = class_names[cls_id] if cls_id < len(class_names) else f"cls_{cls_id}"
247        print(f"{cls_id} ({name}): {correct}/{total} = {acc * 100:.2f}%")
248
249    return overall_accuracy, per_class_accuracy

252def train_model(
253    data_config: Path,
254    epochs: int = DEFAULT_EPOCHS,
255    batch_size: int = 16,
256    project_dir: Path = Path("all_images_training"),
257    run_name: str = "run_2",
258    device: str = "cpu",
259) -> YOLO:
260    """
261    Train a YOLOv8 model using the Ultralytics API.
262
263    This function is optional and provided for completeness. In many
264    workflows, you may choose to train in Colab or elsewhere, and only
265    load the resulting `.pt` weights locally.
266
267    Parameters
268    ----------
269    data_config : Path
270        Path to the YOLO data configuration file (data.yaml).
271    epochs : int, optional
272        Number of training epochs, by default DEFAULT_EPOCHS.
273    batch_size : int, optional
274        Training batch size, by default 16.
275    project_dir : Path, optional
276        Directory where Ultralytics will store run artifacts, by default
277        "all_images_training".
278    run_name : str, optional
279        Subdirectory name for this training run, by default "run_2".
280    device : str, optional
281        Device to train on ("cpu" or "cuda"), by default "cpu".
282
283    Returns
284    -------
285    YOLO
286        The trained YOLO model instance.
287    """
288    # Load a base YOLO model (e.g., YOLOv8 nano variant)
289    model = YOLO("yolov8n.pt")
290
291    results = model.train(
292        data=str(data_config),
293        epochs=epochs,
294        batch=batch_size,
295        save_period=5,
296        device=device,
297        cls=2.0,
298        project=str(project_dir),
299        name=run_name,
300        save=True,
301        exist_ok=True,
302    )
303
304    # Evaluate on the validation set defined in the data config
305    metrics = model.val()
306    print(f"mAP@0.5: {metrics.box.map}")
307
308    return model
